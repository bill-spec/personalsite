set.seed(343)
x1 <- rnorm(10000, 5, 1.5)
x2 <- rnorm(10000, 7, 1.5)
a <- as.data.frame(cbind(x1,x2)) %>% 
  pivot_longer(cols  =c(x1,x2), names_to = "Distribution", values_to = "Value") %>% 
  mutate(Distribution = as.factor(Distribution))
a
